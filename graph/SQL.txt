#include <bits/stdc++.h>
using namespace std;
int  a[5][5];
int main() {
	int n,m,x,y;
    cin>>n>>m;
    //n=nodes m=edges
    for(int i=1;i<m;i++)
    {
        cin>>x>>y;
        a[x][y]=1;
        a[y][x]=1;//undirected
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            cout<<a[i][j]<<" ";
        }
        cout<<endl;
    }
    
    
    	return 0;
}

#include <bits/stdc++.h>
using namespace std;
vector <int> v[5];
int main() {
	int n,m,x,y;
    cin>>n>>m;
    //n=nodes m=edges
    //1 2
    //1->2
    //v[1].pushback(2)
    for(int i=1;i<m;i++)
    {
        cin>>x>>y;
        v[x].push_back(y);
         v[y].push_back(x);//undirected
    }
    for(int i=0;i<n;i++)
    {
        cout<<i<<"->";
        for(int j=0;j<v[i].size();j++)
        {
            cout<<v[i][j]<<" ";
        }
        cout<<endl;
    }
    
    
    	return 0;
}
